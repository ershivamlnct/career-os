% =============================================================================
% HEADER
% =============================================================================

\NewDocumentCommand{\ResumeHeader}
    { m m m m m m }
{
    \begin{center}

    {\NameFont\color{Primary}\textbf{#1}}

    \vspace{4pt}

    {\HeadlineFont\color{Secondary}\textbf{#2}}

    \vspace{5pt}

    {\MetaFont
    \faMapMarker* \hspace{2pt} #3
    \hspace{12pt}
    \faEnvelope \hspace{2pt}
    \href{mailto:#4}{#4}
    \hspace{12pt}
    \faPhone \hspace{2pt} #5
    \hspace{12pt}
    \faLinkedin \hspace{2pt}
    \href{https://#6}{#6}
    }

    \end{center}

    \vspace{\HeaderBottomSpace}

    \SectionDivider
}

% =============================================================================
% SECTION
% =============================================================================

\NewDocumentCommand{\ResumeSection}{m}
{
    \section*{
        \color{SectionTitle}
        \SectionFont
        #1
    }

    \vspace{2pt}
}
